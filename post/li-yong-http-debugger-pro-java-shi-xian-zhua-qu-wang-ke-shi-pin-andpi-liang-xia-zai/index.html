利用HTTP Debugger Pro + JAVA 实现抓取网课视频&批量下载

2021-11-24 / 6 min read

我们都知道Fidder可以实现抓包，但相对配置比较复杂，而使用 HTTP Debugger Pro 这个工具可以傻瓜式抓包，对于小白来说比较友好（没错我就是小白），进入正题：

  1. 下载 HTTP Debugger Pro

    下载中文版简单易懂，双击即可使用

  2. 可以使用过滤器功能对相应进行过滤，比如如果我要爬视频：可以过滤类型为：video/mp4
  3. 打开对应的app或者网站，点击对应的视频，就可以抓到url，如图：

    一条视频往往会有多个响应，应该是为了应对不同的环境，但他的url地址是完全相同的，所以之后做过滤就好了

  4. 把要下载的视频全部点击一遍，全选然后复制url，也可以将所有响应导出为Json格式

  5. 接下来就是下载了，批量下载网上有一些软件，但命名又需要手动命名，所以干脆写了一个java脚本，效果如下：


    源码：
package com.test.application.controller;


import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;
import com.google.gson.JsonArray;
import com.google.gson.JsonObject;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import java.io.*;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLConnection;
import java.text.NumberFormat;
import java.util.*;
import java.util.stream.Collectors;
import java.util.stream.Stream;

/**
 * @Author 海
 * @Time 2021/11/17 19:26
 * @Description:
 */

@RestController
@RequestMapping("/batchDownload")
public class TestController {

    /**
     * 解析JSON
     * @param param
     * @return
     */
    @RequestMapping("/parse")
    public String parse(@RequestBody String param) {
        JSONArray jsonArray = JSONObject.parseArray(param);
        System.out.println("JsonArray的长度为" + jsonArray.size());
        LinkedList<String> list = new LinkedList<>();
        for (int i = 0; i < jsonArray.size(); i++) {
            String url = jsonArray.getString(i);
            list.add(url);
        }
        Collections.reverse(list); // 倒序排列
        return JSONObject.toJSONString(list);
    }

    /**
     * 下载执行
     * @param param
     */
    @RequestMapping("/execute")
    public void test(@RequestBody String param) {
        JSONArray jsonArray = JSONObject.parseArray(param);
        String filePath = "C:\\Users\\82661\\Desktop\\软考视频\\";
        System.out.println("共计"+jsonArray.size()+"个文件待下载");
        for (int i = 0; i < jsonArray.size(); i++) {
            long startTime = System.currentTimeMillis();
            File file = saveUrlAs(jsonArray.getString(i), filePath, (i + ".mp4"), "GET",i+1);
            long endTime = System.currentTimeMillis();
            int remain = jsonArray.size() - 1;
            System.out.println("下载\t" + (i + ".mp4") + "\t用时" + (endTime - startTime) / 1000 + "秒\t" + "文件大小：" + getSize(file) + "\t剩余文件数：" + remain + "个");
        }
    }

    /**
     * 批量下载/命名视频
     * @param url       下载url
     * @param filePath  文件保存路径
     * @param filename  文件名
     * @param method    请求方法
     * @return
     */
    public static File saveUrlAs(String url, String filePath, String filename, String method,int j) {
        //System.out.println("fileName---->"+filePath);
        //创建不同的文件夹目录
        File file = new File(filePath);
        //判断文件夹是否存在
        if (!file.exists()) {
            //如果文件夹不存在，则创建新的的文件夹
            file.mkdirs();
        }
        FileOutputStream fileOut = null;
        HttpURLConnection conn = null;
        InputStream inputStream = null;
        try {
            // 建立链接
            URL httpUrl = new URL(url);
            conn = (HttpURLConnection) httpUrl.openConnection();
            //以Post方式提交表单，默认get方式
            conn.setRequestMethod(method);
            conn.setDoInput(true);
            conn.setDoOutput(true);
            // post方式不能使用缓存
            conn.setUseCaches(false);
            //连接指定的资源
            conn.connect();
            //获取网络输入流
            inputStream = conn.getInputStream();
            BufferedInputStream bis = new BufferedInputStream(inputStream);
            //判断文件的保存路径后面是否以/结尾
            if (!filePath.endsWith("/")) {

                filePath += "/";

            }
            //写入到文件（注意文件保存路径的后面一定要加上文件的名称）
            fileOut = new FileOutputStream(filePath + filename);
            BufferedOutputStream bos = new BufferedOutputStream(fileOut);
            int i = readSize(url);
            System.out.println("待下载文件大小：" + i / 1024 / 1024 + "M");
            byte[] buf = new byte[4096];
            int length = bis.read(buf);
            int sizeRead = 0;
            String now = null;
            //保存文件
            while (length != -1) {
                bos.write(buf, 0, length);
                length = bis.read(buf);

                sizeRead += length;
                // 创建一个数值格式化对象
                NumberFormat numberFormat = NumberFormat.getInstance();
                // 设置精确到小数点后2位
                numberFormat.setMaximumFractionDigits(2);
                String result = numberFormat.format((float) sizeRead / (float) i * 100);
                if (!result.equals(now) && result.length() <= 2) {
                    System.out.println("正在下载第"+j+"个文件：\t" + filename + "\t" + result + "%");
                    now = result;
                }
            }
            bos.close();
            bis.close();
            conn.disconnect();
        } catch (Exception e) {
            e.printStackTrace();
            System.out.println("文件下载失败！！");
        }

        return new File(filePath + filename);
    }

    /**
     * 预读取待下载文件大小
     * @param urls
     * @return
     */
    private static int readSize(String urls) {
        int size = 0;
        try {
            URL url = new URL(urls);
            URLConnection conn = url.openConnection();
            size = conn.getContentLength();
            conn.getInputStream().close();
        } catch (IOException e) {
            e.printStackTrace();
        }
        return size;
    }

    /**
     * 读取已下载文件大小
     * @param f
     * @return
     */
    private static String getSize(File f) {
        FileInputStream fis = null;
        try {
            fis = new FileInputStream(f);
            int available = fis.available();
            int size = available / 1024 / 1024;
            return size + "M";
        } catch (Exception e) {
            System.out.println(e);
        } finally {
            if (null != fis) {
                try {
                    fis.close();
                } catch (IOException e) {
                    System.out.println(e);
                }
            }
        }
        return null;
    }


}
